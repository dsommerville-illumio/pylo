def pylo.vendors.psutil._pssunos.wrap_exceptions (   fun) 
Call callable into a try/except clause and translate ENOENT,
EACCES and EPERM in NoSuchProcess or AccessDenied exceptions.
340 def wrap_exceptions(fun):
341  """Call callable into a try/except clause and translate ENOENT,
342  EACCES and EPERM in NoSuchProcess or AccessDenied exceptions.
343  """
344  @functools.wraps(fun)
345  def wrapper(self, *args, **kwargs):
346  try:
347  return fun(self, *args, **kwargs)
348  except (FileNotFoundError, ProcessLookupError):
349  # ENOENT (no such file or directory) gets raised on open().
350  # ESRCH (no such process) can get raised on read() if
351  # process is gone in meantime.
352  if not pid_exists(self.pid):
353  raise NoSuchProcess(self.pid, self._name)
354  else:
355  raise ZombieProcess(self.pid, self._name, self._ppid)
356  except PermissionError:
357  raise AccessDenied(self.pid, self._name)
358  except OSError:
359  if self.pid == 0:
360  if 0 in pids():
361  raise AccessDenied(self.pid, self._name)
362  else:
363  raise
364  raise
365  return wrapper
