391  That matches "available" column in newer versions of "free".

393  missing_fields = []
394  mems = {}
395  with open_binary('%s/meminfo' % get_procfs_path()) as f:
396  for line in f:
397  fields = line.split()
398  mems[fields[0]] = int(fields[1]) * 1024
399 
400  # /proc doc states that the available fields in /proc/meminfo vary
401  # by architecture and compile options, but these 3 values are also
402  # returned by sysinfo(2); as such we assume they are always there.
403  total = mems[b'MemTotal:']
404  free = mems[b'MemFree:']
405  try:
406  buffers = mems[b'Buffers:']
407  except KeyError:
408  # https://github.com/giampaolo/psutil/issues/1010
409  buffers = 0
410  missing_fields.append('buffers')
411  try:
412  cached = mems[b"Cached:"]
413  except KeyError:
414  cached = 0

416  else:
417  # "free" cmdline utility sums reclaimable to cached.
418  # Older versions of procps used to add slab memory instead.
419  # This got changed in:

421  # 05d751c4f076a2f0118b914c5e51cfbb4762ad8e
422  cached += mems.get(b"SReclaimable:", 0) # since kernel 2.6.19
423 
424  try:

426  except KeyError:
427  try:
428  shared = mems[b'MemShared:'] # kernels 2.4
429  except KeyError:
430  shared = 0
431  missing_fields.append('shared')
432

435  except KeyError:
436  active = 0
437  missing_fields.append('active')
438

441  except KeyError:

445  mems[b"Inact_clean:"] + \
446  mems[b"Inact_laundry:"]
447  except KeyError:
448  inactive = 0
449  missing_fields.append('inactive')

453  except KeyError:
