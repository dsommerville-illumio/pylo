def pylo.vendors.psutil._psaix.wrap_exceptions (   fun) 
Call callable into a try/except clause and translate ENOENT,
EACCES and EPERM in NoSuchProcess or AccessDenied exceptions.
311 def wrap_exceptions(fun):
312  """Call callable into a try/except clause and translate ENOENT,
313  EACCES and EPERM in NoSuchProcess or AccessDenied exceptions.
314  """
315  @functools.wraps(fun)
316  def wrapper(self, *args, **kwargs):
317  try:
318  return fun(self, *args, **kwargs)
319  except (FileNotFoundError, ProcessLookupError):
320  # ENOENT (no such file or directory) gets raised on open().
321  # ESRCH (no such process) can get raised on read() if
322  # process is gone in meantime.
323  if not pid_exists(self.pid):
324  raise NoSuchProcess(self.pid, self._name)
325  else:
326  raise ZombieProcess(self.pid, self._name, self._ppid)
327  except PermissionError:
328  raise AccessDenied(self.pid, self._name)
329  return wrapper
