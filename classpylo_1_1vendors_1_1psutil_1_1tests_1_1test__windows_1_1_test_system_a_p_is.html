155  for ps_part in ps_parts:
156  for wmi_part in wmi_parts:
157  if ps_part.device.replace('\\', '') == wmi_part.DeviceID:
158  if not ps_part.mountpoint:

162  break
163  try:
164  usage = psutil.disk_usage(ps_part.mountpoint)
165  except FileNotFoundError:
166  # usually this is the floppy
167  break
168  self.assertEqual(usage.total, int(wmi_part.Size))
169  wmi_free = int(wmi_part.FreeSpace)
170  self.assertEqual(usage.free, wmi_free)
171  # 10 MB tollerance
172  if abs(usage.free - wmi_free) > 10 * 1024 * 1024:
173  self.fail("psutil=%s, wmi=%s" % (
174  usage.free, wmi_free))

176  else:
177  self.fail("can't find partition %s" % repr(ps_part))
