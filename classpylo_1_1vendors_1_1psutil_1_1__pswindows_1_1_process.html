899  def wait(self, timeout=None):
900  if timeout is None:
901  cext_timeout = cext.INFINITE
902  else:
903  # WaitForSingleObject() expects time in milliseconds.
904  cext_timeout = int(timeout * 1000)
905 
906  timer = getattr(time, 'monotonic', time.time)
907  stop_at = timer() + timeout if timeout is not None else None
908 
909  try:
910  # Exit code is supposed to come from GetExitCodeProcess().
911  # May also be None if OpenProcess() failed with
912  # ERROR_INVALID_PARAMETER, meaning PID is already gone.
913  exit_code = cext.proc_wait(self.pid, cext_timeout)
914  except cext.TimeoutExpired:
915  # WaitForSingleObject() returned WAIT_TIMEOUT. Just raise.
916  raise TimeoutExpired(timeout, self.pid, self._name)
917  except cext.TimeoutAbandoned:
918  # WaitForSingleObject() returned WAIT_ABANDONED, see:
919  # https://github.com/giampaolo/psutil/issues/1224
920  # We'll just rely on the internal polling and return None
921  # when the PID disappears. Subprocess module does the same
922  # (return None):
923  # https://github.com/python/cpython/blob/
924  # be50a7b627d0aa37e08fa8e2d5568891f19903ce/
925  # Lib/subprocess.py#L1193-L1194
926  exit_code = None
927 
928  # At this point WaitForSingleObject() returned WAIT_OBJECT_0,
929  # meaning the process is gone. Stupidly there are cases where
930  # its PID may still stick around so we do a further internal
931  # polling.
932  delay = 0.0001
933  while True:
934  if not pid_exists(self.pid):
935  return exit_code
936  if stop_at and timer() >= stop_at:
937  raise TimeoutExpired(timeout, pid=self.pid, name=self._name)
938  time.sleep(delay)
939  delay = min(delay * 2, 0.04) # incremental delay
