Workaround for https://github.com/giampaolo/psutil/issues/875.

713 def retry_error_partial_copy(fun):
714  """Workaround for https://github.com/giampaolo/psutil/issues/875.
715  See: https://stackoverflow.com/questions/4457745#4457745
716  """
717  @functools.wraps(fun)
718  def wrapper(self, *args, **kwargs):
719  delay = 0.0001
720  times = 33
721  for x in range(times): # retries for roughly 1 second
722  try:
723  return fun(self, *args, **kwargs)
724  except WindowsError as _:
725  err = _
726  if err.winerror == ERROR_PARTIAL_COPY:
727  time.sleep(delay)
728  delay = min(delay * 2, 0.04)
729  continue
730  else:
731  raise
732  else:
733  msg = "%s retried %s times, converted to AccessDenied as it's " \
734  "still returning %r" % (fun, times, err)
735  raise AccessDenied(pid=self.pid, name=self._name, msg=msg)
